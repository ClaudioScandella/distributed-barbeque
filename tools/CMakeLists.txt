
# .:: Mandatory Tools required for the usage of the BarbequeRTRM
# These tools must be:
# - part of the BarbequeTOOLS componet
# - have a name starting by "bbque-"

	DESTINATION ${BBQUE_PATH_BBQ}
	COMPONENT BarbequeTOOLS
	RENAME bbque-plp)
